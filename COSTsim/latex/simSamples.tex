\HeaderA{simSamples}{Simulation of data sets}{simSamples}
\keyword{methods}{simSamples}
\begin{Description}\relax
This function simulates data sets for the evaluation and comparison of the estimation methods
\end{Description}
\begin{Usage}
\begin{verbatim}
simSamples(obj, ...)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{obj}] a 'simData' object 
\item[\code{...}] Further arguments 
\end{ldescription}
\end{Arguments}
\begin{Value}
It returns a simData object with the \code{sample} slots containing the simulated data sets
\end{Value}
